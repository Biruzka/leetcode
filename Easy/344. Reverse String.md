# 344. Reverse String

## Complexity
- Time complexity: *O(n)*

- Space complexity: *O(1)*

## TypeScript
```
/**
 Do not return anything, modify s in-place instead.
 */
function reverseString(s: string[]): void {
    var tmp;
    var i = 0;
    var j = s.length - 1;
    while (j>i){
        tmp = s[i];
        s[i] = s[j];
        s[j] = tmp;
        i++;
        j--;
    }
};
```
## C#
```
public class Solution {
    public void ReverseString(char[] s) {
        int i = 0;
        int j = s.Length - 1;
        while (j > i)
        {
            char tmp = s[i];
            s[i] = s[j];
            s[j] = tmp;
            i++;
            j--;
        }
    }
}
```
